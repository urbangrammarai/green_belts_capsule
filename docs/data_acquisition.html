def read_clip(irow, p='signatures.gpkg'):
    i, row = irow
    geom = row.geometry
    out = geopandas.read_file(
        p, mask=geom
    ).clip(geom)
    for col in ['LA_Code', 'LA_Name', 'GB_Name']:
        out[col] = row[col]
    return out
